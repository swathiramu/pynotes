Python Tuple

Thu 15 January 2026

Creating a Tuple

empty = ()
print(empty)

()

single= (10)
print(single)

10

num = (1, 2, 3, 4)
print(num)

(1, 2, 3, 4)

mixed = (10, "Python", True)
print(mixed)

(10, 'Python', True)

Accessing Tuple Elements

colors = ("red", "green", "blue")
print(colors[0])

red

print(colors[-1])

blue

print(colors[-2])

green

print(colors[1])

green

Slicing Tuples

values = (0, 1, 2, 3, 4)
print(values[1:4])

(1, 2, 3)

values = (0, 1, 2, 3, 4)

print(values[0:3])

(0, 1, 2)

Tuple Packing and Unpacking

person = ("Swathiii", 21, "Engineer")
name, age, profession = person

print(name) 

Swathiii

print(age) 

21

print(profession)

Engineer

Nested Tuples

data = ((1, 2), (3, 4), (5, 6))
print(data[1])

(3, 4)

print(data[2])

(5, 6)

print(data[1][0])

3

print(data[2][1])

6

Tuple Length and Membership

num = (10, 20, 30)
print(len(num))

3

numbers = (10, 20, 30, 78, 90)

print(len(numbers))

5

print(20 in numbers) 

True

print(60 in numbers) 

False

Iterating Through a Tuple

languages = ("Python", "Java", "Go")
for lang in languages:
    print(lang)

Python
Java
Go

Converting Between Tuples and Lists

values = (1, 2, 3)

list1 = list(values)
tuple1 = tuple(list1)

print(list1)   
print(tuple1) 

[1, 2, 3]
(1, 2, 3)
