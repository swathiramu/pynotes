Python Numbers and Mathematics

Numeric Data Types in Python

int = 10
print(type(int))

<class 'int'>

float = 3.14
print(type(float))

<class 'float'>

complex = 2 + 5j
print(type(complex))

<class 'complex'>

Basic Mathematical Operations

a = 12
b = 5
print(a + b) 

17

print(a - b)

7

print(a * b)

60

print(a / b)

2.4

Floor Division and Modulus

x = 17
y = 5

print(x // y) 

3

print(x % y)

2

base = 2
exponent = 4

res = base ** exponent
print(res)

16

Math Module Basics

import math
print(math.sqrt(25))

5.0

print(math.ceil(4.3))

5

print(math.floor(4.9))

4

Trigonometric Functions

import math
a = math.radians(30)
print(math.sin(a)) 

0.49999999999999994

print(math.cos(a))

0.8660254037844387

print(math.tan(a)) 

0.5773502691896257

Rounding Numbers

val = 3.5678
print(round(val)) 

4

print(round(val, 2)) 

3.57

Absolute and Sign Functions

num = -25
print(abs(num)) 

25

print(math.copysign(10, -1))

-10.0

Random Number Generation

import random
print(random.randint(1, 30))

16

print(random.uniform(1.0, 5.0))

2.4907373255645604

Statistical Operations

import statistics
num = [10, 20, 30, 40, 50]
print(statistics.mean(num))   

30

print(statistics.median(num)) 

30

print(statistics.stdev(num))  

15.811388300841896
