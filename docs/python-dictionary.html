Python Dictionary

Thu 15 January 2026

Creating a Dictionary

dict = {}
print(dict)

{}

student = {"name": "Alice", "age": 22, "grade": "A"}
print(student)

{'name': 'Alice', 'age': 22, 'grade': 'A'}

Accessing Dictionary Values

emp = {"id": 186, "name": "Swathi", "role": "Developer"}
print(emp)

{'id': 186, 'name': 'Swathi', 'role': 'Developer'}

emp = {"id": 186, "name": "Swathi", "role": "Developer"}
print(emp["name"])

Swathi

emp = {"id": 186, "name": "Swathi", "role": "Developer"}
print(emp.get("role"))

Developer

Adding and Updating Dictionary Entries

prof = {"username": "admin"}
prof["email"] = "swathiramu28@gmail.com"
prof["username"] = "admin"
print(prof)

{'username': 'admin', 'email': 'swathiramu28@gmail.com'}

Removing Items from a Dictionary

data = {"a": 1, "b": 2, "c": 3}
removed = data.pop("b")
del data["a"]
print(data)    

{'c': 3}

print(removed)

2

Dictionary Length and Membership

set = {"theme": "dark", "version": 1.0}
print(len(set))  

2

print("theme" in set) 

True

Iterating Through Dictionaries

user = {"name": "Swathi", "age": 21, "city": "London"}

for key in user:
    print(key, ":", user[key])

for key, value in user.items():
    print(key, "=>", value)

name : Swathi
age : 21
city : London
name => Swathi
age => 21
city => London

Dictionary Methods

data = {"x": 10, "y": 20}
print(data.keys()) 

dict_keys(['x', 'y'])

Merging Dictionaries

dict1 = {"a": 1, "b": 2}
dict2 = {"c": 3, "d": 4}
merged = dict1 | dict2
print(merged)

{'a': 1, 'b': 2, 'c': 3, 'd': 4}

Nested Dictionaries

employee = {"id": 101,"personal": {"name": "John","email": "john@example.com"}}
print(employee["personal"]["name"])

John

Dictionary Comprehension

squa = {x: x**2 for x in range(5)}
print(squa) 

{0: 0, 1: 1, 2: 4, 3: 9, 4: 16}
