Using the nonlocal Keyword

def outer():
    count = 0

    def inner():
        nonlocal count
        count += 1
        print(count)

    inner()

LEGB Rule (Scope Resolution Order)

Python resolves variables using the LEGB rule:

Local

Enclosing

Global

Built-in

x = "Global"

def outer():
    x = "Enclosing"

    def inner():
        x = "Local"
        print(x)

    inner()

outer()

Local

Built-in Scope

Scope Inside Loops

for i in range(3):
    loop = i

print(loop)

Practical Example of Scope Management

total = 100

def calculate():
    total = 50

    def adjust():
        nonlocal total
        total += 10

    adjust()
    return total

print(calculate())

60

print(total)  

100
