Creating a Set

set = {1, "Python", 3.5, True}

Automatic Removal of Duplicates

num = {1, 2, 2, 3, 3, 4, 6, 7, 7, 8 , 8}
print(num) 

{1, 2, 3, 4, 6, 7, 8}

f = {"apple", "banana"}
f.add("orange")
print(f)

{'banana', 'orange', 'apple'}

Adding Multiple Elements with update()

c = {"red", "green"}
c.update(["blue", "yellow"])

print(c)

{'green', 'blue', 'red', 'yellow'}

Removing Elements from a Set

num = {10, 20, 30, 40, 50}
num.remove(20)
print(num)

{50, 40, 10, 30}

num = {10, 20, 30, 40, 50}
num.discard(50)
print(num)

{20, 40, 10, 30}

num = {10, 20, 30, 40, 50}
num.discard(10)
print(num)

{50, 20, 40, 30}

num = {10, 20, 30, 40, 50}
num.discard(30)
print(num)

{50, 20, 40, 10}

True

print(100 not in data)

True

False

Mathematical Set Operations

A = {1, 2, 3, 4}
B = {3, 4, 5, 6}
print(A | B)

{1, 2, 3, 4, 5, 6}

print(A & B)  

{3, 4}

print(A - B)

{1, 2}

print(A ^ B)

{1, 2, 5, 6}

A = {1, 2, 3}
B = {1, 2}

print(B.issubset(A)) 

True

print(A.issuperset(B))

True

print(A.issubset(B)) 

False

Converting Sets to Other Types

set = {1, 2, 3}
listv = list(set)
print(listv) 

[1, 2, 3]

tuplev = tuple(set)
print(tuplev)

(1, 2, 3)
